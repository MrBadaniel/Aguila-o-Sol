character_event = { # Pillaging Repeating Event
	id = woodrunner.4800
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_realm_province = {
			any_province_holding = {
				owner = { character = ROOT }
				has_holding_modifier = recently_forested_the_land
				NOT = { has_holding_modifier = stop_freeing_wildlife }
			}
		}
	}
	
	immediate = {
		character_event = { id = woodrunner.4800 days = 180 }
		add_character_modifier = {
			modifier = freeing_wildlife_cooldown
			hidden = yes
			months = 6
		}
		any_realm_province = {
			any_province_holding = {
				limit = {
					owner = { character = ROOT }
					has_holding_modifier = recently_forested_the_land
					NOT = { has_holding_modifier = stop_freeing_wildlife }
				}
				add_holding_modifier = {
					modifier = recently_forested_the_land
					months = 7
				}
			}
			any_province_holding = {
				limit = {
					owner = { character = ROOT }
					has_holding_modifier = recently_forested_the_land
					NOT = { has_holding_modifier = stop_freeing_wildlife }
					has_siege = no
				}
	
				if = {
					limit = { NOT = { holding_type = tribal } }
					ROOT = {
						wealth = 50
					}
					custom_tooltip = {
						text = gain_tech_custom_tooltip
						hidden_tooltip = {
							ROOT = {
								random_list = {
									10 = { military_techpoints = 5 }
									10 = { economy_techpoints = 5 }
									10 = { culture_techpoints = 5 }
								}
							}
						}
					}
				}
	
				if = {
					limit = { holding_type = tribal }
					ROOT = { 
						wealth = 25 
					}
				}
	
				destroy_random_building = THIS
				destroy_random_building = THIS
	
				location = {
					add_province_modifier = {
						modifier = rampaging_ranger
						years = 10
					}
				}

				location = {
					add_province_modifier = {
						modifier = liberated_lands_past_tense
						years = 10
					}
				}
	
				if = {
					limit = {
						NOT = { has_any_building = yes }
						owner = {
							character = ROOT
						}
					}
					location = {
						province_event = { id = woodrunner.4810 days = 1 }
						save_event_target_as = target_location
						ROOT = { character_event = { id = woodrunner.4814 days = 2 random = 2 } }
					}
					destroy_settlement = THIS	
				}
			}
		}
	}
}

province_event = { # Clear the 'Rampaging Nomads'-Modifier when province is empty, also prosperity mods
	id = woodrunner.4810
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				NOT = { is_capital = yes }
				NOT = {
					any_province_holding = {
						OR = {
							holding_type = city
							holding_type = temple
							holding_type = castle
							holding_type = tribal
						}
					}
				}
			}
			remove_province_modifier = rampaging_ranger
			remove_province_modifier = prosperity_modifier_1
			remove_province_modifier = prosperity_modifier_2
			remove_province_modifier = prosperity_modifier_3
			remove_province_modifier = depopulated_1
			remove_province_modifier = depopulated_2
			remove_province_modifier = depopulated_3
		}
	}

}
