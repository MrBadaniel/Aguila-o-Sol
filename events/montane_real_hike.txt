# Drunkard/hedonist at holy site
character_event = {
	id = montane_pilgrimage.5066
	desc = EVTDESC_montane_pilgrimage_5066
	picture = GFX_evt_feast

	is_triggered_only = yes

	option = {
		name = EVTOPTA_montane_pilgrimage_5066

		if = {
			limit = {
				lifestyle_traits < 1
				NOT = { has_dlc = "Way of Life" }
			}

			random = {
				chance = 30
				add_trait = hedonist
			}
		}

		random = {
			chance = 30
			add_trait_drunkard_effect = yes
		}

		add_character_modifier = {
			name = good_times
			years = 2
		}

		ai_chance = { factor = 80 }
	}

	option = {
		name = EVTOPTB_montane_pilgrimage_5066

		piety = 10

		random = {
			chance = 30
			add_trait_stressed_effect = yes
		}

		ai_chance = { factor = 20 }
	}

	after = {
		character_event = {
			id = montane_pilgrimage.5090 # Stay a little longer then go home
			days = 10
		}
	}
}
