		true_father = {
			has_character_flag = st_john_baptiste_father
			NOT = { has_character_flag = was_st_john_baptiste_father }
		}
	}
	
	immediate = {
		add_trait = strong
		add_trait = quick
		remove_trait = weak

		remove_trait = inbred

		remove_trait = slow
		remove_trait = stutter
		remove_trait = ugly

			limit = {
				is_female = yes
			}

		}
		mother = {
			clr_character_flag = st_john_baptiste_mother
