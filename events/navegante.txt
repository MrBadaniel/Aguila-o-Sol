namespace = navegante

character_event = { #event to add iconoclast raid variable for when game starts, character converts to purificador or reaches adulthood etc
	id = navigant.1
	hide_window = yes
	
	min_age = 16
	
	is_triggered_only = yes
	
	is_playable = yes
	
	religion_group = cult_of_saints
	
	trigger = {
			OR = {  
				religion = navegante
					AND = {
					trait = marian_stella_maris
					religion = sacred_heart
				}
			}
		NOT = {
			check_variable = { which = forced_conversions value >= 1 }
		}
	}
	
	immediate = {
		set_variable = { which = forced_conversions value = 0 }
	}
}

character_event = { #event that gives the traits
	id = navigant.3
	desc = { 
		trigger = { 
			NOR = {
				trait = levita
				trait = aaronita
				trait = melquisedequita
			}
		} 
		text = EVTDESC_navigant_3A
	}
	desc = { 
		trigger = { 
			trait = levita
		}
		text = EVTDESC_navigant_3B
	}
	desc = { 
		trigger = { 
			trait = aaronita
		}
		text = EVTDESC_navigant_3C
	}
	picture = GFX_evt_iconoclasts_iconoclasming
	
	hide_from = yes
	
	is_triggered_only = yes
	
	min_age = 16
	religion_group = cult_of_saints
	
	
	trigger = {
			OR = {  
				religion = navegante
					AND = {
					trait = marian_stella_maris
					religion = sacred_heart
				}
			}
		OR = {
			AND = {
				NOR = {
					trait = levita
					trait = aaronita
					trait = melquisedequita
				}
				check_variable = { which = forced_conversions value = 5 }
			}
			AND = {
				trait = levita
				check_variable = { which = forced_conversions value = 10 }
			}
			AND = {
				trait = aaronita
				check_variable = { which = forced_conversions value = 15 }
			}
		}
	}
	
	option = {
        name = EVTOPTA_navigant_3
		trigger = {
			NOR = {
				trait = levita
				trait = aaronita
				trait = melquisedequita
			}
			check_variable = { which = forced_conversions value = 10 }
		}
		add_trait = levita
	}
	
	option = {
        name = EVTOPTB_navigant_3
		trigger = {
			trait = levita
			check_variable = { which = forced_conversions value = 25 }
		}
		remove_trait = levita
		add_trait = aaronita
	}
	
	option = {
        name = EVTOPTC_navigant_3
		trigger = {
			trait = aaronita
			check_variable = { which = forced_conversions value = 50 }
		}
		remove_trait = aaronita
		add_trait = melquisedequita
	}
}
