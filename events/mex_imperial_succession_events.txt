### Lose Tapado trait by being exiled from realm, dying, or becoming emperor
character_event = {
	id = destape.3
	desc = DEBUG
	is_triggered_only = yes # Triggered by on_death, on_yearly_pulse
	hide_window = yes
	
	trigger = {
		OR = {
			trait = tapado
			trait = destapado
			# trait = se_movio
		}
	}
	
	immediate = {
		remove_trait = tapado
		remove_trait = destapado
		# remove_trait = se_movio
	}
	
	option = {
		name = DEBUG
	}
}

### Destape Ceremony Decision
narrative_event = {
	id = destape.4
	title = "TITLE_DESTAPE"
	desc = EVTDESC_destape_4
	picture = GFX_evt_powerful_ruler
	
	is_triggered_only = yes

	option = {
		name = EVTOPTA_destape_4 # ready the ceremony
		set_character_flag = holding_destape
		hidden_tooltip = {
			narrative_event = { id = destape.7 days = 15 } # ceremony start Benemérito
			any_realm_character = {
				limit = {
					is_adult = yes
					NOT = { trait = incapable }
					prisoner = no	
				}
				clr_character_flag = destape_has_ended
				narrative_event = { id = destape.5 }
			}
		}
	}
	option = {
		name = EVTOPTB_destape_4 # I'm not ready
		hidden_tooltip = {
			letter_event = {
				id = destape.12
			}
		}
	}
}

### Destape Begins
narrative_event = {
	id = destape.6
	title = "TITLE_DESTAPE"
	desc = EVTDESC_destape_6
	picture = GFX_evt_shadowy_cabal

	is_triggered_only = yes
	
	#war = no
	
	trigger = {
		NOR = {
			has_character_flag = holding_destape
			has_character_flag = host_destape_begins
		}
		any_realm_character = {
			NOT = {
				has_character_flag = destape_begins
			}
			has_character_flag = attending_destape
		}
	}
			
	option = {
		name = EVTOPTA_destape_6 # time for the dedazo
		# set_character_flag = host_destape_begins
		hidden_tooltip = {
			set_character_flag = destape_begins
			# ROOT = { character_event = { id = destape.10 days = 5 } }
			any_courtier_or_vassal = {
				limit = {
					trait = tapado
				}
				character_event = { id = destape.8 }
			}
		}
	}
}

### Tapado is unveiled - sent to Tapado
character_event = {
	id = destape.8
	title = "TITLE_DESTAPE"
	desc = EVTDESC_destape_8
	picture = GFX_evt_society_joining
	portrait = FROM

	is_triggered_only = yes
	
	trigger = {
		trait = tapado
	}
	
	immediate = {
		FROMFROMFROM = { 
			long_character_event = { 
				id = destape.10 days = 5
			} 
		}
	}
	
	option = {
		name = EVTOPTA_destape_8
		hidden_tooltip = {	
			remove_trait = tapado
			add_trait = destapado
			any_realm_character = {
				limit = {
					has_character_flag = attending_destape
					NOR = { 
						trait = tapado
						trait = destapado
						character = FROM						
					}
				}
				narrative_event = { id = destape.9 }
			}
		}
	}
}
