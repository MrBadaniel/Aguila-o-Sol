	is_triggered_only = yes
	
	option = { #heck yeah
		name = EVTOPTA_destape_1

			factor = 100
		}
	}
	option = { #well, I don't wanna
		name = EVTOPTB_destape_1

			#FROM = {
				add_character_modifier = {

 					hidden = yes
				}
			#}
		}
		hidden_tooltip = {
			FROMFROM = {
				character_event = { id = destape.3 }
			}
		}
		ai_chance = {
			factor = 1

				factor = 70

					trait = blinded

	}

### Lose Tapado trait by being exiled from realm
character_event = {
	id = destape.6
	desc = DEBUG
	is_triggered_only = yes # Triggered by on_death, on_yearly_pulse
	hide_window = yes
	
	trigger = {
		OR = {
			trait = tapado
			trait = destapado
			trait = se_movio
		}
	}
	
	immediate = {
		remove_trait = tapado
		remove_trait = destapado
		remove_trait = se_movio
	}
	
	option = {
		name = DEBUG
	}
}
