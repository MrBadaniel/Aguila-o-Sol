add_cenobitic_trait_effect = {
	if = {
		limit = { is_female = yes }
			add_trait = canoness
	}
	if = {
		limit = { is_female = no }
			add_trait = canon
		
	}
}

add_cenobitic_trait_effect_married = {
	if = {
		limit = { is_female = yes }
			add_trait = deaconess
	}
	if = {
		limit = { is_female = no }
			add_trait = deacon
	}
}

set_desert_name_effect = {
	if = {
		limit = { is_female = yes }
		set_special_character_title = MOUNTAIN_MOTHER_TITLE
	}
	if = {
		limit = { is_female = no }
		set_special_character_title = MOUNTAIN_FATHER_TITLE
	}
}

}
