has_mexican_culture_trigger = {
	OR = {
		culture_group = mexicano
		culture_group = norteno
	}
}

has_indigenous_mexican_culture_trigger = {
	OR = {
		culture_group = mesoamerican
		culture_group = mayan
		culture_group = taracahita
	}
}

# Mexican Imperial Govt. is used for Mexican Empire and kingdom of Mexico
mexican_government_trigger = {
	OR = {
		AND = {
			top_liege = {
				OR = {
					title = e_mexico
					title = k_mexico
				}
			}
			OR = { 
				title = e_mexico
				title = k_mexico
			}
			holder_scope = {
				OR = {
					culture_group = mexicano
					culture_group = norteno
					AND = {
						religion_group = cult_of_saints
						OR = {
							culture_group = mesoamerican
							culture_group = mayan
							culture_group = taracahita
						}
					}
				}
			}
		}
	}
}
